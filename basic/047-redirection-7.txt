# Here-document (<< EOF) feeds lines to stdin of cat.
# "cat file" would also read file, but stdin is appended after file argument.
# "1> file2" redirects stdout of cat to file2.
# So file2 will contain: contents of "file" followed by line "hello" (with newline).
cat file << EOF 1> file2
hello
EOF

# End of file
