# Set PS1 permanently by adding to ~/.bashrc (or ~/.profile if login shell reads it).
# Prompt should look like:
# [gamma 3 12:21] ~/tmp$
#
# PS1 parts:
# \\h = hostname
# \\! = history number
# \\A = time HH:MM
# \\w = current working directory
PS1="[\h \! \A] \w\\$ "
