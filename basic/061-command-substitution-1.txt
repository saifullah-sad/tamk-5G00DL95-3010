# $(...) runs in a subshell:
# 1) ls -la runs and outputs listing to stdout
# 2) pipe sends it to wc -l
# 3) wc -l counts lines and outputs a number N
# 4) command substitution replaces $(...) with N
# Final output:
# The Result of this command is N
echo "The Result of this command is $(ls -la | wc -l) "
